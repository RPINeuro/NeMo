103 void neuronReceiveMessage(neuronState *st, tw_stime time, Msg_Data *m,
104 tw_lp *lp){
105  bool willFire = false;
106  st->firedLast = false;
107  //save the previous state of the neuron:
108  st->savedLastLeakTime = st->lastLeakTime;
109  st->savedLastActiveTime = st->lastActiveTime;
110  st->savedMembranePot = st->membranePot;
111  //random fn call state management.
112  unsigned long startCount = lp->rng->count;
113 
114  switch (m->eventType) {
115  case SYNAPSE_OUT:
116  integrateSynapse(m->localID, st, lp);
117 
118  //next, we will check if a heartbeat message should be sent
119  if(st->receivedSynapseMsgs == 0) {
120  sendHeartbeat(st, lp, time);
121  }
122  st->receivedSynapseMsgs ++;
123 
124 
125  break;
126 
127  case NEURON_HEARTBEAT:
128  st->receivedSynapseMsgs = 0;
129  //set up drawn random number for the heartbeat.
130  if(st->thresholdPRNMask != 0)
131  st->drawnRandomNumber = tw_rand_integer(lp->rng, 0, st->thresholdPRNMask);
132 
133  //Currently operates - leak->fire->(reset)
134  st->doLeak(st, time);
135  willFire = neuronShouldFire(st,lp);
136  if(willFire){
137  neuronFire(st, time, lp);
138  st->fireCount ++;
139  }
140 
141  neuronPostIntegrate(st, time, lp, willFire);
142  //stats collection
143  st->SOPSCount ++;
144  st->lastActiveTime = tw_now(lp);
145  break;
146 
147  default:
148  //Error condition - non-valid input.
149  break;
150  }
151 
152  //store the random calls in the message:
153  m->rndCallCount = lp->rng->count - startCount;
154 
155  st->rcvdMsgCount ++;

214 void neuronPostIntegrate(neuronState *st, tw_stime time, tw_lp *lp, bool willFire){
215  if(willFire){ // neuron will/did fire:
216  st->doReset(st);
217  } else if (st->membranePot < -1 * (st->negativeThreshold * st->negThresReset + (st->negativeThreshold + st->drawnRandomNumber))){
218  //sanity variables for the formulaic reset/bounce instead of calling functions:
219  _threshT B = st->negativeThreshold;
220  int K = st->negThresReset;
221  int G = st->resetMode;
222  _randT n = st->drawnRandomNumber;
223  _voltT R = st->resetVoltage;
224  _voltT V = st->membranePot;
225  st->membranePot = (-(B*K) + (-(DT(G)) * R +
226  DT(G-1) * (V + (B + n)) +
227  DT(G-2) * V) * (1-K));
228 
229  }
